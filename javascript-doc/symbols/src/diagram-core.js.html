162 /**
163  * @description
164  * <p>We extend the prototype of all functions with the function createDelegate. This method allows
165  * us to change the scope of a function to "this".</p>
166  *
167  * <p>This is useful when attaching listeners to jQuery events like click or mousemove as jQuery normally uses
168  * $(this) to reference the source of the event. When using the createDelegate method, this will point to the
169  * object that you want to reference with this.</p>
170  *
171  * <p>Source: <a href="http://stackoverflow.com/questions/520019/controlling-the-value-of-this-in-a-jquery-event">
172  *     Stackoverflow
173  * </a></p>
174  *
175  * @param {Object} scope The scope which you want to apply.
176  */
177 Function.prototype.createDelegate = function(scope) {
178     var fn = this;
179     return function() {
180         // Forward to the original function using 'scope' as 'this'.
181         return fn.apply(scope, arguments);
182     }
183 }

237 /**
238  * @description
239  * Retrieve all elements that have a specific class. This function is used in combination with svg.
240  *
241  * @param {Object} namespace The svgweb namespace that will be used. In most cases it will be svgns
242  * (not a string but instead a global).
243  * @param {String} theClass the class that the element should have
244  * @param {String} [tagName] The elements tag. This can be used to increase the performance
245  * @return {Object[]} Elements from the namespace.
246  */
247 document.getElementsByClassNS= function(namespace, theClass, tagName) {
248 
249     if (tagName == undefined) {
250         tagName = "*";
251     }
252 
253     var allElements = document.getElementsByTagNameNS(namespace, tagName);
254 
255     var result = [];
256 
257     for (i = 0; i < allElements.length; i++) {
258 
259         var currentElement = allElements[i];
260         var classesOfCurrentElement = currentElement.className.baseVal.split(" ");
261 
262         for(j = 0; j < classesOfCurrentElement.length; j++) {
263             if (classesOfCurrentElement[j] == theClass) {
264                 result[result.length] = currentElement;
265                 break;
266             }
267         }
268     }
269 
270     return result;
271 }

285 /**
286  * @description
287  * This function measures the size of a given text with a specified css.
288  *
289  * @param {String} text The text from which you want to meassure the size
290  * @param {Object} css The css which should be used for meassuring
291  * @return {Object} An object with width and height properties that represent the text dimensions.
292  */
293 odr.meassureTextDimensions = function(text, css) {
294     var div = document.createElement('div');
295     document.body.appendChild(div);
296     $(div).css(css);
297

299

318 /**
319  * @description
320  * <p>Round <i>value</i> to a multiple of <i>roundTo</i></p>
321  *
322  * @param {Number} value The value which you want to round
323  * @param {Number} roundTo Round to the nearest multiple of this value
324  */
325 odr.round = function(value, roundTo) {
326     var modResult = value % roundTo;
327 
328     if(modResult == 0) {
329         return value;
330     } else if (modResult >= (roundTo / 2)) {
331         return value + (roundTo - modResult);
332     } else {
333         return value - modResult;
334     }
335 }

342 /**
343  * @description
344  * Just list {@link odr._round} but it just rounds up to a multiple of <i>roundTo</i>.
345  *
346  * @param {Number} value The value which you want to round
347  * @param {Number} roundTo Round to the nearest multiple of this value
348  * @see odr._round
349  */
350 odr.roundUp = function(value, roundTo) {
351     var modResult = value % roundTo;
352 
353     if(modResult == 0) {
354         return value;
355     }
356 
357     return value + (roundTo - (modResult));
358 }

383 /**
384  * @description
385  * <p>Move all marked shapes</p>
386  *
387  * <p>odr.vars.markedElements is used for performance reasons.</p>
388  *
389  * @param {Number} x The number of pixel that each marked element should be moved in horizontal direction
390  * @param {Number} y The number of pixel that each marked element should be moved in vertical direction
391  * @param {odr.Drawable} exclude Exclude this item from moving. Normally this should be that item that is dragged
392  * as otherwise it would be moved twice.
393  */
394 odr.moveMarkedShapes = function(x, y, exclude) {
395     var root = odr.canvas();
396 
397     // wait with redrawing the SVG
398     var suspendID = root.suspendRedraw(5000);
399 
400     for(var key in odr.vars.markedElements) {
401         var element = odr.vars.markedElements[key];
402 
403         if (element != exclude) {
404             element.move(x, y);
405         }
406     }
407 
408     // enable redrawing
409     root.unsuspendRedraw(suspendID);
410 }

421 /**
422  * @description
423  * <p>Create a new arrow with default attributes and add it to the definitions of the svg.</p>
424  *
425  * @return {Element} The created element
426  */
427 odr.arrow = function(id) {
428     var marker = document.createElementNS(svgns, "marker");
429 
430     if (id != undefined) {
431         marker.id = id;
432     }
433 
434     for(var attributeName in odr.settings.line.arrow.attributes) {
435         marker.setAttribute(attributeName, odr.settings.line.arrow.attributes[attributeName]);
436     }
437 
438     marker.setAttribute("class", odr.settings.line.arrow["class"])
439 
440     var path = document.createElementNS(svgns, "path");
441     path.setAttribute("d", "M0,0L10,5L0,10z");
442     marker.appendChild(path);
443 
444     var defs = document.getElementsByTagNameNS(svgns, "defs")[0];
445     defs.appendChild(marker);
446 
447     return marker;
448 };

496                 $(this).removeClass(odr.translation.titleClass);
497             });
498 
499             // translate static button values

501                 var translation = odr.translation.text[$(this).val()];
502                 $(this).val(translation);
503                 $(this).removeClass(odr.translation["class"]);
504             });
505 
506             // translate static alternative image texts
507             $("img." + odr.translation["class"]).each(function() {
508                 var translation = odr.translation.text[$(this).attr("alt")];

510                 $(this).removeClass(odr.translation["class"]);

512 
513             // translate static html text elements
514             $("." + odr.translation["class"]).each(function() {
515                 var translation = odr.translation.text[$(this).text()];
516                 $(this).text(translation);
517                 $(this).removeClass(odr.translation["class"]);
518             });

553 /**
554  * @description
555  * Make sure that the size of the SVG canvas is big enough for all lines
556  */
557 odr.assertSvgSize = function() {
558     var root = odr.canvas();
559 
560     var width = 0, height = 0;
561     
562     for(var key in odr.vars.shapesThatDetermineCanvasSize) {
563         var shape = odr.vars.shapesThatDetermineCanvasSize[key];
564         var bottomRight = shape.bottomRight();
565 
566         if (shape instanceof odr.Handle || shape.visible()) {
567             width = Math.max(width, bottomRight.x);
568             height = Math.max(height, bottomRight.y);
569         }
570     }
571 
572     width += odr.settings.svg.padding.right;
573     height += odr.settings.svg.padding.bottom;
574 
575     // wait with redrawing the SVG
576     var suspendID = root.suspendRedraw(5000);
577 
578     root.setAttribute("width", width);
579     root.setAttribute("height", height);
580     root.setAttribute("viewBox", "0 0 " + width + " " + height);
581 
582     // enable redrawing
583     root.unsuspendRedraw(suspendID);
584 };

612 /**
613  * @description
614  * This function will be called when the user clicks on the body element.
615  *
616  * @param {jQueryEvent} e The jQuery event
617  */
618 odr._selectionStart = function(e) {
619     if (!$(e.target).is("body, svg")) {
620         return;
621     }
622 
623     odr.vars.lasso = $("#" + odr.settings.lasso.id);
624 
625     odr.vars.lasso.show();
626     odr.vars.lasso.initialX = e.pageX;
627     odr.vars.lasso.initialY = e.pageY;
628     odr.vars.lasso.css("left", e.pageX);
629     odr.vars.lasso.css("top", e.pageY);
630     odr.vars.lasso.css("width", 0);
631     odr.vars.lasso.css("height", 0);
632 
633     if (!e.ctrlKey) {
634         odr.clearSelectedElements();
635     }
636 
637     $("body").mousemove(odr._selectionResize);
638     $("body").mouseup(odr._selectionEnd);
639 
640     // returning false as otherwise the browser will try to select the content of the document
641     return false;
642 };

649 /**
650  * @private
651  *
652  * @description
653  * This function will be called when the user clicked on the body or svg element and moved the mouse
654  *
655  * @param {jQueryEvent} e The jQuery event
656  */
657 odr._selectionResize = function(e) {
658     var x1 = odr.vars.lasso.initialX;
659     var x2 = e.pageX;
660 
661     var y1 = odr.vars.lasso.initialY;
662     var y2 = e.pageY;
663 
664     var minX = Math.min(x1, x2);
665     var minY = Math.min(y1, y2);
666 
667     odr.vars.lasso.css("left", minX);
668     odr.vars.lasso.css("top", minY);
669     odr.vars.lasso.css("width", Math.max(x1, x2) - minX);
670     odr.vars.lasso.css("height", Math.max(y1, y2) - minY);
671 
672     // returning false as otherwise the browser will try to select the content of the document
673     return false;
674 };

690 odr._selectionEnd = function(e) {
691     $("body").unbind("mousemove", odr._selectionResize);

693

695     var top = parseInt(odr.vars.lasso.css("top").removeNonNumbers());
696     var width = odr.vars.lasso.width();
697     var height = odr.vars.lasso.height();
698 
699     odr.selectElements(left, top, left + width, top + height);
700 
701     odr.vars.lasso.hide();
702 
703     // returning false as otherwise the browser will try to select the content of the document
704     return false;
705 };

730 /**
731  * @description
732  * Select all elements in odr.vars.shapesThatDetermineCanvasSize that fall into the given region.
733  *
734  * @param {Number} minX The minimum x coordinate
735  * @param {Number} minY The minimum y coordinate
736  * @param {Number} maxX The maximal x coordinate
737  * @param {Number} maxY The maximal y coordinate
738  */
739 odr.selectElements = function(minX, minY, maxX, maxY) {
740 
741     for(var key in odr.vars.shapesThatDetermineCanvasSize) {
742         var element = odr.vars.shapesThatDetermineCanvasSize[key];
743 
744         var topLeft = element.topLeft();
745         var bottomRight = element.bottomRight();
746 
747         if (topLeft.x >= minX && topLeft.y >= minY && bottomRight.x <= maxX && bottomRight.y <= maxY) {
748             element.marked(!element.marked());
749         }
750     }
751 };

790 /**

834             odr.user.automaticallyAlignDragHandles = true;
835             off.addClass(odr.settings.menu.bottom.selectedClass);
836             on.removeClass(odr.settings.menu.bottom.selectedClass);
837             ask.removeClass(odr.settings.menu.bottom.selectedClass);

842         }
843 
844         $(this).dialog("close");
845     };

1119 /**
1120  * @description
1121  * Taken from http://snipplr.com/view/799/get-url-variables/
1122  *
1123  * @return {Object} An object that contains all request parameters in the form of "parameterName : value".
1124  */
1125 odr.getUrlVars = function() {
1126     var vars = [], hash;
1127     var hashes = window.location.href.slice(window.location.href.indexOf('?') + 1).split('&');
1128     for(var i = 0; i < hashes.length; i++) {
1129         hash = hashes[i].split('=');
1130         vars.push(hash[0]);
1131         vars[hash[0]] = hash[1];
1132     }
1133     return vars;
1134 }

1154  * ###############################################################################################################
1155  *                                              Maintaining the session
1156  */
1157 /**
1158  * @description
1159  * This method is used to keep up the session. It could happen that a edits / views a visualization for longer then
1160  * the session timeout length. Therefore the server is "pinged" continuously to maintain the session.
1161  *
1162  * @param {Boolean} continuous Whether the ping should only be done once or multiple times.
1163  */
1164 odr.maintainSession = function(continuous) {
1165     if (continuous == undefined) {
1166         continuous = true;
1167     }
1168 
1169     $.get(odr.settings.request.translation);
1170     $.ajax({
1171         url: odr.settings.request.sessionMaintaining,
1172         error: function(request, status, error) {
1173             odr.popup.showError("Could not maintain session: " + status, true);
1174         },
1175         success : function() {
1176             if (continuous) {
1177                 setTimeout(odr.maintainSession, odr.settings.sessionMaintaining);
1178             }
1179         }
1180     });
1181 };
1182 
1183 odr.ready(odr.maintainSession);
