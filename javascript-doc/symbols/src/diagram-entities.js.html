 17 /**
 18  * @class
 19  * A helper class that implements the <a href="http://martinfowler.com/eaaCatalog/registry.html">Registry pattern</a>.
 20  * 
 21  * @description
 22  * Every entity automatically registers itself with the registry so that the entity can be retrieved by it's id.
 23  */
 24 odr.Registry = function() {
 25     this._items = {};
 26 }
 27 odr.Registry.prototype = {
 28     _items : null,
 29     /**
 30      * @param {odr.DrawableItem} item The item you whish to add to the Registry. The key by which the item can later on
 31      * be removed or get is the result of the {@link odr.DrawableItem#id()} call.
 32      */
 33     add : function(item) {
 34         this._items[item.id()] = item;
 35     },
 36     /**
 37      * @param {String|Number} itemId
 38      * @return {odr.DrawableItem}
 39      */
 40     get : function(itemId) {
 41         return this._items[itemId];
 42     },
 43     /**
 44      * @param {{String|Number}} itemId
 45      */
 46     remove : function(itemId) {
 47         delete this._items[itemId];
 48     }
 49 }
 50 
 51 odr.bootstrap(function() {
 52     odr.registry = new odr.Registry();
 53 });

 68 /**
 69  * @constructor
 70  *
 71  * @class
 72  * Base class for all entities that can be drawn. A class that inherits from this class states that it can be
 73  * drawn and that it knows how to draw itself.
 74  *
 75  * @abstract

 77  * @description
 78  * Every drawable automatically registers itself with the {@link odr.Registry} so that the entity can be

 87 
 88     this._listener = {};
 89 
 90     for(var listenerType in odr.Drawable.listener) {
 91         this._listener[odr.Drawable.listener[listenerType]] = {};
 92     }

365             listenerCollection[listener].callWithParams(params);
366         }
367 
368         return this;
