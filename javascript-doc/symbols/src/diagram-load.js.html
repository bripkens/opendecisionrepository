  9 odr.ready(function() {
 10     // analyse request parameter
 11     var projectId = odr.vars.requestParameter[odr.settings.request.parameter.projectId];
 12 
 13     if (isNaN(projectId)) {
 14         odr.popup.showError("Invalid project id: " + projectId);
 15         return;
 16     }
 17 
 18     var relationship = odr.vars.requestParameter[odr.settings.request.parameter.relationshipView];
 19     var chronological = odr.vars.requestParameter[odr.settings.request.parameter.chronologicalView];
 20     var stakeholder = odr.vars.requestParameter[odr.settings.request.parameter.stakeholderView];
 21 
 22     var viewHandler = null;
 23     var viewParameter = null;
 24 
 25     if (relationship != undefined) {
 26         viewHandler = odr.handleRelationshipView;
 27         viewParameter = odr.settings.request.parameter.relationshipView;
 28     } else if (chronological != undefined) {
 29         viewHandler = odr.handleChronologicalView;
 30         viewParameter = odr.settings.request.parameter.chronologicalView;
 31     } else if (stakeholder != undefined) {
 32         viewHandler = odr.handleStakeholderView;
 33         viewParameter = odr.settings.request.parameter.stakeholderView;
 34     } else {
 35         odr.popup.showError("Invalid viewpoint");
 36         return;
 37     }
 38 
 39     odr.vars.requestedViewpoint = viewParameter;
 40 
 41     var parameter = {};
 42     parameter[odr.settings.request.parameter.projectId] = projectId;
 43     parameter[viewParameter] = true;
 44 
 45     $.ajax({
 46         url : odr.settings.request.dataProvider,
 47         data : parameter,
 48         dataType : "json",
 49         timeout : 2000,
 50         async : false,
 51         error : function(data, textStatus, errorThrown) {
 52             odr.popup.showError("Error while retrieving data from the server: " + textStatus + " /// " + errorThrown);
 53         },
 54         success : viewHandler
 55     });
 56 });

 65 /**
 66  * @description
 67  * Handle the data which has been loaded from the server. The data must have the correct form for a relationship view
 68  *
 69  * @param {Object} data The data from the server
 70  */
 71 odr.handleRelationshipView = function(data) {
 72     $("#toRelationshipView").hide();
 73 
 74     odr.vars.json = data;
 75 
 76     var root = odr.canvas();
 77     var suspendID = root.suspendRedraw(5000);
 78 
 79     for(var i = 0; i < data.Nodes.length; i++) {
 80         var nodeJson = data.Nodes[i];
 81 
 82         var node = odr.addNode(nodeJson.Version.Decision.Name,
 83             nodeJson.Version.State.StatusName,
 84             nodeJson.Version.State.StatusName,
 85             nodeJson.Visible);
 86 
 87         node.json = nodeJson;
 88         node.position(nodeJson.X, nodeJson.Y);
 89         node.additionalInformationFunction(odr.showAdditionalDecisionDetails);
 90 
 91         if (nodeJson.Width != 0 && nodeJson.Height != 0) {
 92             node.size(nodeJson.Width, nodeJson.Height);
 93         }
 94         
 95         node.addClass(odr.settings.node.additionalClasses.roundCorners);
 96         node.addClass(odr.settings.node.additionalClasses.decision);
 97 
 98         odr.vars.allDecisionNodes[nodeJson.Version.Id] = node;
 99     }
100 
101 
102     for(var i = 0; i < data.Associations.length; i++) {
103         var associationJson = data.Associations[i];
104 
105         var association = new odr.Association();
106         association.json = associationJson;
107         association.label(associationJson.Relationship.Type.Name);
108 
109         if (associationJson.LabelX == 0) {
110             associationJson.LabelX = 50;
111         }
112 
113         if (associationJson.LabelY == 0) {
114             associationJson.LabelY = 50;
115         }
116 
117         association.labelPosition(associationJson.LabelX, associationJson.LabelY);
118         association.source(odr.vars.allDecisionNodes[associationJson.Relationship.Source.Id]);
119         association.target(odr.vars.allDecisionNodes[associationJson.Relationship.Target.Id]);
120 
121         association.loadHandles(associationJson.Handles);
122 
123         odr.vars.allAssociations[associationJson.Id] = association;
124     }
125 
126     root.unsuspendRedraw(suspendID);
127 };

137 /**
138  * @description
139  * Handle the data which has been loaded from the server. The data must have the correct form for a chronological view
140  *
141  * @param {Object} data The data from the server
142  */
143 odr.handleChronologicalView = function(data) {
144     $("#toChronologicalView").hide();
145 
146     odr.vars.json = data;
147 
148     var root = odr.canvas();
149     var suspendID = root.suspendRedraw(5000);
150 
151     for(var i = 0; i < data.Nodes.length; i++) {
152         var nodeJson = data.Nodes[i];
153 
154         var label, status, statusToShow, additionalInformationFunction;
155 
156         if (nodeJson.Version != null) {
157             label = nodeJson.Version.Decision.Name;
158             status = nodeJson.Version.State.StatusName;
159             statusToShow = nodeJson.Version.State.StatusName;
160             additionalInformationFunction = odr.showAdditionalDecisionDetails;
161         } else {
162             label = nodeJson.Iteration.Name;
163             status = nodeJson.Iteration.EndDate;
164             statusToShow = odr.translation.text["label.iteration"];
165             additionalInformationFunction = odr.showAdditionalIterationDetails;
166         }
167         
168 
169         var node = odr.addNode(label,
170             status,
171             statusToShow,
172             nodeJson.Visible);
173 
174 
175         node.json = nodeJson;
176         node.position(nodeJson.X, nodeJson.Y);
177         node.additionalInformationFunction(additionalInformationFunction);
178 
179         if (nodeJson.Width != 0 && nodeJson.Height != 0) {
180             node.size(nodeJson.Width, nodeJson.Height);
181         }
182 
183         if (nodeJson.Disconnected == true) {
184             node.addClass(odr.settings.node.additionalClasses.disconnected);
185         }
186 
187 
188         if (nodeJson.Version != null) {
189             node.addClass(odr.settings.node.additionalClasses.roundCorners);
190             node.addClass(odr.settings.node.additionalClasses.decision);
191             odr.vars.allDecisionNodes[nodeJson.Version.Id] = node;
192         } else {
193             node.addClass(odr.settings.node.additionalClasses.iteration);
194             odr.vars.allIterationNodes[nodeJson.Iteration.Id] = node;
195         }
196     }
197 
198 
199     for(var i = 0; i < data.Associations.length; i++) {
200         var associationJson = data.Associations[i];
201 
202         var association = new odr.Association();
203         association.json = associationJson;
204 
205         if (associationJson.SourceIteration != null) {
206             association.source(odr.vars.allIterationNodes[associationJson.SourceIteration.Id]);
207         } else {
208             association.source(odr.vars.allDecisionNodes[associationJson.SourceVersion.Id]);
209         }
210 
211         if (associationJson.TargetIteration != null) {
212             association.target(odr.vars.allIterationNodes[associationJson.TargetIteration.Id]);
213         } else {
214             association.target(odr.vars.allDecisionNodes[associationJson.TargetVersion.Id]);
215         }
216 
217         
218         association.loadHandles(associationJson.Handles);
219 
220         odr.vars.allAssociations[associationJson.Id] = association;
221     }
222 
223     root.unsuspendRedraw(suspendID);
224 };
