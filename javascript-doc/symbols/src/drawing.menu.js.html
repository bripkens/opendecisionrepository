 59 /**
 60  * @description
 61  * Attach listeners to all the zoom menu buttons and images
 62  */
 63 odr.prepareZoom = function() {
 64     j("#" + odr.zoomMenu.inputButton).click(odr.handleScaleInput);
 65 
 66     j("#" + odr.zoomMenu.inButton).click(function() {
 67         odr.scale(odr.scale() + 0.1);
 68         j("#" + odr.zoomMenu.inputField).val(Math.round(odr.scale() * 100));
 69     });
 70 
 71     j("#" + odr.zoomMenu.outButton).click(function() {
 72         odr.scale(odr.scale() - 0.1);
 73         j("#" + odr.zoomMenu.inputField).val(Math.round(odr.scale() * 100));
 74     });
 75 
 76     j("#" + odr.zoomMenu.inputField).keyup(function(e) {
 77         if(e.keyCode == 13) {
 78             odr.handleScaleInput();
 79         }
 80     });
 81 
 82     j("#" + odr.zoomMenu.inputField).val(odr.scale() * 100);
 83 }
 84 
 85 /**
 86  * @description
 87  * This functions handles scale level changes, i.e. when the user enters something into the zoom level text box
 88  * and hits enter or presses the button below.
 89  */
 90 odr.handleScaleInput = function() {
 91     var scaleInput = j("#" + odr.zoomMenu.inputField);
 92 
 93     var newScale = scaleInput.val();
 94 
 95     newScale = parseFloat(newScale);
 96 
 97     if (isNaN(newScale)) {
 98         scaleInput.addClass(odr.zoomMenu.errorClass);
 99         return;
100     }
101 
102     scaleInput.removeClass(odr.zoomMenu.errorClass);
103 
104     odr.scale(newScale / 100);
105 }

164 /**
165  * @description
166  * Save the positions of all nodes as well as handles by copying the data into the object that was retrieved
167  * from the server.
168  */
169 odr._saveAll = function() {
170     odr.setSaveText();
171     odr.loadPopupIcon();
172     odr.showWaitAnimation();
173 
174 
175     for(var i in odr._allRectangles) {
176         var currentRectangle = odr._allRectangles[i];
177         var value = currentRectangle.value();
178         
179         value.X = currentRectangle.x();
180         value.Y = currentRectangle.y();
181         value.Visible = currentRectangle.visible();
182     }
183 
184     for(var i in odr._allAssociations) {
185         var currentAssociation = odr._allAssociations[i];
186         var value = currentAssociation.value();
187 
188         if (currentAssociation._labelPosition != null) {
189             value.LabelX = currentAssociation._labelPosition.x;
190             value.LabelY = currentAssociation._labelPosition.y;
191         }
192 
193         var handles = currentAssociation.handles();
194 
195         var handlesForValue = [];
196 
197         for (var k = 0; k < handles.length; k++) {
198             var currentHandle = handles[k];
199 
200             handlesForValue[handlesForValue.length] = {
201                 X : currentHandle.x(),
202                 Y : currentHandle.y()
203             };
204         }
205 
206         value.Handles = handlesForValue;
207     }
208 
209     var targetUrl = j("#externalVarTargetUrl").text();
210 
211     var projectIdParameter = j("#externalVarProjectIdParameterName").text();
212     var dataParameter = j("#externalVarDataParameterName").text();
213     var viewpointParameter = j("#externalVarViewpointParameterName").text();
214     
215     var parameter = {};
216     parameter[projectIdParameter] = j("#externalVarProjectId").text();
217     parameter[dataParameter] = odr._createJSONString(odr._requestedData);
218     parameter[viewpointParameter] = true;
219 
220 
221     j.ajax({
222         url : targetUrl,
223         data : parameter,
224         dataType : "json",
225         type : "POST",
226         success : function(data, textStatus, XMLHttpRequest) {
227             odr.setSuccessText();
228             odr.successPopupIcon();
229             j(".clickBlocker").hide();
230             odr.hideWaitAnimation(4000);
231         },
232         error : function(data, textStatus, errorThrown) {
233             odr.setErrorText();
234             odr.errorPopupIcon();
235         }
236     });
237 };

240 /**
241  * @description
242  * The following function could be replaced by a JavaScript JSON serializer like
243  * JSON2 or the jquery json plugin. But for some reason, every quote is escaped.
244  *
245  * http://groups.google.com/group/google-gson/browse_thread/thread/98249309c8455a91
246  *
247  * One good thing about this approach is that only the absolutely necessary information is sent to the server.
248  *
249  * @param {Object} data The original JSON data that should be transformed
250  */
251 odr._createJSONString = function(data) {
252     var json = [];
253 
254     json.push('{"Id":',
255         data.Id,
256         ',"Nodes":[');
257 
258 
259     for (var i = 0; i < data.Nodes.length; i++) {
260         var currentNode = data.Nodes[i];
261 
262         json.push('{"Id":',
263             currentNode.Id,
264             ',"Visible":',
265             currentNode.Visible,
266             ',"X":',
267             currentNode.X,
268             ',"Y":',
269             currentNode.Y,
270             '}'
271             );
272 
273         if (i+1 != data.Nodes.length) {
274             json.push(',');
275         }
276     }
277 
278     json.push('],"Associations":[');
279 
280     for(var i = 0; i < data.Associations.length; i++) {
281         var currentAssociation = data.Associations[i];
282 
283         json.push('{"Id":',
284             currentAssociation.Id,
285             ',"LabelX":',
286             currentAssociation.LabelX,
287             ',"LabelY":',
288             currentAssociation.LabelY,
289             ',"Handles":[');
290 
291         for(var k = 0; k < currentAssociation.Handles.length; k++) {
292             var currentHandle = currentAssociation.Handles[k];
293 
294             json.push('{"X":',
295                 currentHandle.X,
296                 ',"Y":',
297                 currentHandle.Y,
298                 '}');
299 
300             if (k+1 != currentAssociation.Handles.length) {
301                 json.push(',');
302             }
303         }
304 
305         json.push("]}");
306 
307         if (i+1 != data.Associations.length) {
308             json.push(',');
309         }
310     }
311     
312     json.push(']}');
313 
314     return json.join("");
315 }

333 /**
334  * @description
335  * Use this utility functionality to add nodes. It has several benefits as it will make sure that the node will be
336  * added to the nodes menu and it will also add the node to the status menu.
337  *
338  * @param {String} name The name which should be shown for this node
339  * @param {String} stereotype The stereotype that will be visible in the status menu
340  * @param {String} stereotypeToShow The stereotype that will be shown above the name of the decision in the node and in the decisions menu
341  * @param {Number} x The x position
342  * @param {Number} y The y position
343  * @param {Boolean} [visible] Whether this node is visible. Default = false
344  * @param {Boolean} [round] Whether the node has round corners. Default = true
345  * @param {String} [extraClasses] Some extra classes that you want to attach {@link odr.Rectangle#extraClasses}. Default = ""
346  * @return {odr.Rectangle} The new node. The node has been painted to the canvas!
347  */
348 odr.addNode = function(name, stereotype, stereotypeToShow, x, y, visible, round, extraClasses) {
349     if (visible == undefined) {
350         visible = false;
351     }
352 
353     if (round == undefined) {
354         round = true;
355     }
356 
357     if (extraClasses == undefined) {
358         extraClasses = "";
359     }
360 
361     var rectangle = new odr.Rectangle();
362     rectangle.x(x);
363     rectangle.y(y);
364     rectangle.label(name);
365     rectangle.stereotype(stereotypeToShow);
366     rectangle.round(round);
367     rectangle.extraClasses(extraClasses);
368     rectangle.paint();
369 
370 
371     var container = j(document.createElement("div")).
372     addClass("decisionSelector").
373     appendTo("div.decisions div.overflowDecisions").
374     append(j(document.createElement("span")).text(name)).
375     append(j(document.createElement("span")).text(stereotype).addClass("stereotype"));
376 
377     var showText = j("span#decisionShowButtonText").text();
378     var hideText = j("span#decisionHideButtonText").text();
379 
380     container.html(container.html() + '<input type="button" class="hide" value="' + hideText + '"/>');
381 
382     var button = container.children("input");
383 
384     button.click(function() {
385         if (button.hasClass("hide")) {
386             button.removeClass("hide");
387             button.addClass("show");
388             button.val(showText);
389 
390             rectangle.visible(false);
391         } else {
392             button.removeClass("show");
393             button.addClass("hide");
394             button.val(hideText);
395 
396             rectangle.visible(true);
397 
398             odr.assertContainerSize();
399         }
400     });
401 
402     rectangle.visibility(function() {
403         if (rectangle.visible()) {
404             button.removeClass("show");
405             button.addClass("hide");
406             button.val(hideText);
407         } else {
408             button.removeClass("hide");
409             button.addClass("show");
410             button.val(showText);
411         }
412     }, "menuListener");
413 
414 
415 
416 
417 
418 
419 
420     if (odr._nodes[stereotype] == undefined) {
421         odr._nodes[stereotype] = [];
422         odr._newStereotype(stereotype, odr._nodes[stereotype]);
423     }
424 
425     var items = odr._nodes[stereotype];
426 
427     items[items.length] = rectangle;
428 
429     rectangle.visibility(function() {
430         var allVisible = true;
431         var allInvisible = true;
432 
433         for(var i = 1; i < items.length; i++) {
434             if (items[i].visible()) {
435                 allInvisible = false;
436             } else {
437                 allVisible = false;
438             }
439         }
440 
441         if (allVisible) {
442             items[0].attr("checked", "checked");
443             items[0].attr("checked", true);
444             items[0].get().checked = true;
445             
446         } else if (allInvisible) {
447             items[0].attr("checked", false);
448             items[0].removeAttr("checked");
449             items[0].get().checked = false;
450         }
451     }, "stereotypeListener");
452 
453     rectangle.visible(visible);
454 
455     return rectangle;
456 }

459 odr._newStereotype = function(name, items) {
460     var container = j(document.createElement("div")).
461     appendTo("div.status div.overflowStatus").
462     append(j(document.createElement("span")).text(name));
463 
464     container.html('<input type="checkbox" checked="true" id="stereotypeCheckbox' + name + '"/>' + container.html());
465 
466     var checkbox = container.children("input");
467 
468     items[0] = checkbox;
469 
470     checkbox.change(function() {
471 
472         var checked = checkbox.is(":checked");
473         
474         for(var i = 1; i < items.length; i++) {
475             items[i].visible(checked);
476         }
477     });
478 }

493 /**
494  * @description
495  * Show the wait animation popup (actually this popup can also be used to represent errors or the success status).
496  *
497  * @param {String|Number} [animationSpeed] Either "fast" or "slow" for 200 respectively 600 milliseconds or
498  * the time in milliseconds. Default speed is 200 milliseconds.
499  *
500  */
501 odr.showWaitAnimation = function(animationSpeed) {
502     if (animationSpeed == undefined) {
503         animationSpeed = "fast";
504     }
505 
506     var windowWidth = j(document).width();
507     var windowHeight = j(document).height();
508     var popupHeight = j(".loader").height();
509     var popupWidth = j(".loader").width();
510 
511     j(".clickBlocker").css({
512         "height": windowHeight,
513         "width": windowWidth
514     });
515 
516     j(".loader").css({
517         "top": windowHeight/2-popupHeight/2,
518         "left": windowWidth/2-popupWidth/2
519     });
520 
521     j(".clickBlocker").fadeIn(animationSpeed);
522     j(".loader").fadeIn(animationSpeed);
523 }
524 
525 /**
526  * @description
527  * Hide the small popup
528  *
529  * @param {String|Number} [animationSpeed] Either "fast" or "slow" for 200 respectively 600 milliseconds or
530  * the time in milliseconds. Default speed is 200 milliseconds.
531  */
532 odr.hideWaitAnimation = function(animationSpeed) {
533     if (animationSpeed == undefined) {
534         animationSpeed = "fast";
535     }
536 
537     j(".loader").fadeOut(animationSpeed);
538     j(".clickBlocker").fadeOut(animationSpeed);
539 }
