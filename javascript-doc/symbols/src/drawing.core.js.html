270 /**
271  * @description
272  * <p>We extend the prototype of all functions with the function createDelegate. This method allows
273  * us to change the scope of a function to "this".</p>
274  *
275  * <p>This is useful when attaching listeners to jQuery events like click or mousemove as jQuery normally uses
276  * $(this) to reference the source of the event. When using the createDelegate method, this will point to the
277  * object that you want to reference with this.</p>
278  *
279  * <p>Source: <a href="http://stackoverflow.com/questions/520019/controlling-the-value-of-this-in-a-jquery-event">
280  *     Stackoverflow
281  * </a></p>
282  *
283  * @param {Object} scope The scope which you want to apply.
284  */
285 Function.prototype.createDelegate = function(scope) {
286     var fn = this;
287     return function() {
288         // Forward to the original function using 'scope' as 'this'.
289         return fn.apply(scope, arguments);
290     }
291 }

348 /**
349  * @description
350  * <p>Make sure that the element which you specified by elementId fit on the canvas. If it doesn't, resize the canvas.</p>
351  *
352  * <p>If you don't specify a parameter, then make sure that all {@link odr.Shape}s fit on the canvas.</p>
353  *
354  * @param {String|Number} elementId The id for which you want to make sure that it fits on the canvas.
355  */
356 odr.assertContainerSize = function(elementId) {
357     if (elementId == -1) {
358         return;
359     }
360 
361 
362     var container = odr._svgContainer.children("svg");
363 
364     var containerWidth = 0;
365     var containerHeight = 0;
366 
367     var selector = undefined;
368 
369     if (elementId) {
370         selector = "#" + elementId;
371         containerWidth = container.attr("width") - odr.svgContainer.padding.x;
372         containerHeight = container.attr("height") - odr.svgContainer.padding.y;
373     } else {
374         selector = "." + odr.rectangleSettings["class"] + ", ." + odr.handleSettings["class"] + ", ." + odr.associationSettings.text["class"];
375     }
376 
377     j(selector).each(function() {
378         var currentElement = odr.registry.get(j(this).attr("id"));
379 
380         if (currentElement == undefined) {
381             currentElement = odr.registry.get(j(this).attr("id").removeNonNumbers());
382         }
383 
384         var lowerRightCorner = currentElement.bottomRight();
385 
386         lowerRightCorner.x *= odr._scale.level;
387         lowerRightCorner.y *= odr._scale.level;
388 
389         if (lowerRightCorner.x > containerWidth) {
390             containerWidth = lowerRightCorner.x;
391         }
392 
393         if (lowerRightCorner.y > containerHeight) {
394             containerHeight = lowerRightCorner.y;
395         }
396     });
397 
398     containerWidth += odr.svgContainer.padding.x
399     containerHeight += odr.svgContainer.padding.y;
400 
401     container.attr("width", containerWidth);
402     container.attr("height", containerHeight);
403     container.attr("viewBox", "0 0 " + containerWidth + " " + containerHeight);
404     var parent = odr._svgContainer;
405     parent.css({
406         width :  containerWidth,
407         height : containerHeight
408     });
409 }

432 /**
433  * @description
434  * <p>Round <i>value</i> to a multiple of <i>roundTo</i></p>
435  *
436  * @param {Number} value The value which you want to round
437  * @param {Number} roundTo Round to the nearest multiple of this value
438  */
439 odr._round = function(value, roundTo) {
440     var modResult = value % roundTo;
441 
442     if(modResult == 0) {
443         return value;
444     } else if (modResult >= (roundTo / 2)) {
445         return value + (roundTo - modResult);
446     } else {
447         return value - modResult;
448     }
449 }
450 
451 /**
452  * @description
453  * Just list {@link odr._round} but it just rounds up to a multiple of <i>roundTo</i>.
454  *
455  * @param {Number} value The value which you want to round
456  * @param {Number} roundTo Round to the nearest multiple of this value
457  * @see odr._round
458  */
459 odr._roundUp = function(value, roundTo) {
460     var modResult = value % roundTo;
461 
462     if(modResult == 0) {
463         return value;
464     }
465 
466     return value + (roundTo - (modResult));
467 }

469 /**
470  * @description
471  * <p>Snap the position of an {@link odr.Shape} to the grid. The grid dimensions are defined in the
472  * {@link odr.grid} settings.</p>
473  *
474  * <p>The position will be rounded up or down.</p>
475  *
476  * @param {odr.Shape} element The element which you want to snap to the grid
477  */
478 odr.snapPosition = function(element) {
479     if (element.x() != undefined) {
480         element.x(odr._round(element.x(), odr.grid.width));
481     }
482 
483     if (element.y() != undefined) {
484         element.y(odr._round(element.y(), odr.grid.height));
485     }
486 }
487 
488 /**
489  * @description
490  * <p>Snap the position and size of an {@link odr.Shape} to the grid. The grid dimensions are defined in the
491  * {@link odr.grid} settings.</p>
492  *
493  * <p>The size will only be rounded up.</p>
494  *
495  * @param {odr.Shape} element The element which you want to snap to the grid
496  * @see odr.snapPosition
497  */
498 odr.snap = function(element) {
499     odr.snapPosition(element);
500 
501     if (element.width() != undefined) {
502         element.width(odr._roundUp(element.width(), odr.grid.width * 2));
503     }
504 
505     if (element.height() != undefined) {
506         element.height(odr._roundUp(element.height(), odr.grid.height * 2));
507     }
508 }

535 /**
536  * @description
537  * <p>This function will be called when the user clicks on a node. It is responsible for starting the process
538  * of dragging a node.</p>
539  *
540  * @param {jQuery event} e The jQuery mouse event
541  */
542 odr._dragStart = function(e) {
543     if(e.ctrlKey) {
544         return false;
545     }
546 
547     var container = odr._svgContainer.children("svg");
548 
549     var containerWidth = container.attr("width") * odr.drag.extensionsMultiplier
550     var containerHeight = container.attr("height") * odr.drag.extensionsMultiplier
551     container.attr("width", containerWidth);
552     container.attr("height", containerHeight);
553     container.attr("viewBox", "0 0 " + containerWidth + " " + containerHeight);
554     odr._svgContainer.css({
555         width :  containerWidth,
556         height : containerHeight
557     });
558 
559     var button = e.button;
560     odr._dragging.elementToDrag[button] = j(this)
561     odr._dragging.previousEvent[button] = e;
562     
563     j("body").mousemove(odr._drag);
564     j("body").addClass(odr.grid["class"]);
565 
566     if ((odr._dragging.itemToDrag[button] = odr.registry.get(j(this).attr("id"))) == null) {
567         odr._dragging.itemToDrag[button] = odr.registry.get(j(this).attr("id").removeNonNumbers());
568     }
569     
570     odr._dragging.itemToDrag[button].dragStart();
571 
572     return false;
573 }

575 /**
576  * @description
577  * <p>This function handles the actual dragging, i.e. when the user clicked on a node and is now moving the mouse.</p>
578  *
579  * <p>This function also makes sure that the container sizes adapts itself to the position and size of the dragged
580  * node. This function also contains the functionality that enables the user to drag a node to the border of the
581  * </p>
582  *
583  * @param {jQuery event} e The jQuery mouse event
584  */
585 odr._drag = function(e) {
586     var button = e.button;
587     if (odr._dragging.previousEvent[button] == undefined) {
588         j("body").unbind("mousemove");
589         return false;
590     }
591 
592     var oldX = odr._dragging.itemToDrag[button].x();
593     var oldY = odr._dragging.itemToDrag[button].y();
594 
595     var deltaX = (e.pageX - odr._dragging.previousEvent[button].pageX) * (1 / odr._scale.level);
596     var deltaY = (e.pageY - odr._dragging.previousEvent[button].pageY) * (1 / odr._scale.level);
597 
598     odr._dragging.itemToDrag[button].x(oldX + deltaX);
599     odr._dragging.itemToDrag[button].y(oldY + deltaY);
600     odr._dragging.previousEvent[button] = e;
601 
602     odr._dragging.itemToDrag[button].dragging();
603 
604     odr.assertContainerSize(odr._dragging.itemToDrag[button].extendedId());
605 
606 
607     var viewportWidth = j(window).width();
608     var viewportHeight = j(window).height();
609     var scrollLeft = j(window).scrollLeft();
610     var scrollTop = j(window).scrollTop();
611     var lowerRightCorner = odr._dragging.itemToDrag[button].bottomRight();
612     var topLeftCorner = odr._dragging.itemToDrag[button].topLeft();
613 
614     var scrollX = 0;
615     var scrollY = 0;
616 
617     var doScrollRight = (lowerRightCorner.x * odr._scale.level) - scrollLeft > viewportWidth - odr.scroll.margin.right;
618     var doScrollLeft = (topLeftCorner.x * odr._scale.level) - scrollLeft < odr.scroll.margin.left;
619     var doScrollDown = (lowerRightCorner.y * odr._scale.level) > viewportHeight - odr.scroll.margin.down - scrollTop;
620     var doScrollUp = (topLeftCorner.y * odr._scale.level) < scrollTop + odr.scroll.margin.up;
621 
622     if (doScrollRight) {
623         scrollX = odr.scroll.speed.right;
624     } else if (doScrollLeft) {
625         scrollX = odr.scroll.speed.left;
626     }
627 
628     if (doScrollUp) {
629         scrollY = odr.scroll.speed.up;
630     } else if (doScrollDown) {
631         scrollY = odr.scroll.speed.down;
632     }
633 
634     window.scrollBy(scrollX, scrollY);
635 
636     return false;
637 }

640 /**
641  * @description
642  * <p>When the user releases the mouse button (while dragging), this function is called.</p>
643  *
644  * <p>
645  * This function hides the grid and detaches temporarily bound listeners. In addition, {@link odr.assertContainerSize}
646  * is called to make sure that the canvas size is adapted.
647  * </p>
648  *
649  * @param {jQuery event} e The jQuery mouse event
650  */
651 odr._dragStop = function(e) {
652     var button = e.button;
653 
654     if (odr._dragging.itemToDrag[button] == undefined || e.ctrlKey) {
655         return false;
656     }
657 
658     j("body").removeClass(odr.grid["class"]);
659 
660     j("body").unbind("mousemove");
661     odr._dragging.previousEvent[button] = undefined;
662 
663     odr._dragging.itemToDrag[button].dragEnd();
664 
665     odr.assertContainerSize();
666 
667     return false;
668 }

686 /**
687  * @description
688  * <p>Scale the association to a new level or retrieve the current scale level by calling the method without a
689  * parameter.</p>
690  *
691  * <p>The default scale level is 1. The reduce the scale level by 10% set the scale level to 0.9.</p>
692  *
693  * @param {Number} [newScale] The new scale level
694  * @return {Number} Always the current scale level. When setting a new scale level, the new level is returned.
695  */
696 odr.scale = function(newScale) {
697     if (!newScale) {
698         return odr._scale.level;
699     }
700 
701     odr._scale.level = newScale;
702 
703     var transformAttribute = "scale(" + odr._scale.level + ")";
704 
705     for (var i = 0; i < odr.groups.length; i++) {
706         j("#" + odr.groups[i]).attr("transform", transformAttribute);
707     }
708 
709     odr.assertContainerSize();
710 
711     return odr._scale.level;
712 }

727 /**
728  * @description
729  * Call this function before exporting. This function will reduce the white borders at the sides of the image.
730  */
731 odr.beforeExport = function() {
732 
733     var container = odr._svgContainer.children("svg");
734 
735     var x = null;
736     var y = null;
737     var width = 0;
738     var height = 0;
739 
740 
741     j("." + odr.rectangleSettings["class"] + ", ." + odr.handleSettings["class"] + ", ." + odr.associationSettings.text["class"]).each(function() {
742         var currentElement = odr.registry.get(j(this).attr("id"));
743 
744         if (currentElement == undefined) {
745             currentElement = odr.registry.get(j(this).attr("id").removeNonNumbers());
746         }
747 
748         var lowerRightCorner = currentElement.bottomRight();
749         lowerRightCorner.x *= odr._scale.level;
750         lowerRightCorner.y *= odr._scale.level;
751 
752         var topLeftCorner = currentElement.topLeft();
753         topLeftCorner.x *= odr._scale.level;
754         topLeftCorner.y *= odr._scale.level;
755 
756         if (lowerRightCorner.x > width) {
757             width = lowerRightCorner.x;
758         }
759 
760         if (lowerRightCorner.y > height) {
761             height = lowerRightCorner.y;
762         }
763 
764         if (x == null || topLeftCorner.x < x) {
765             x = topLeftCorner.x;
766         }
767 
768         if (y == null || topLeftCorner.y < y) {
769             y = topLeftCorner.y;
770         }
771     });
772 
773     width += odr.svgContainer.padding.x
774     height += odr.svgContainer.padding.y;
775 
776     x -= odr.svgContainer.padding.x
777     y -= odr.svgContainer.padding.y;
778 
779     container.attr("width", width - x);
780     container.attr("height", height - y);
781 
782     var viewBox = x + " " + y + " " + (parseInt(width) - parseInt(x)) + " " + (parseInt(height) - parseInt(y));
783 
784     container.attr("viewBox", viewBox);
785 }

810 /**
811  * @description
812  * This function measures the size of a given text with a specified css.
813  *
814  * @param {String} text The text from which you want to meassure the size
815  * @param {Object} css The css which should be used for meassuring
816  * @return {Object} An object literal with width and height properties that represent the text dimensions.
817  */
818 odr.meassureTextDimensions = function(text, css) {
819     var div = document.createElement('div');
