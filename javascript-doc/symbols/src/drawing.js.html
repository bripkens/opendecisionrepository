 10 odr.ready(function() {
 11     odr.setLoadingText();
 12     odr.loadPopupIcon();
 13     odr.showWaitAnimation();
 14 
 15     var requestError = j("#externalVarRequestError").text();
 16     if (requestError == "true") {
 17         return;
 18     }
 19 
 20     var projectId = j("#externalVarProjectId").text();
 21     var requestUrl = j("#externalVarRequestUrl").text();
 22     var projectIdParameter = j("#externalVarProjectIdParameterName").text();
 23     var viewpointParameter = j("#externalVarViewpointParameterName").text();
 24 
 25     var parameter = {};
 26     parameter[projectIdParameter] = projectId;
 27     parameter[viewpointParameter] = true;
 28 
 29     var handlerFunction = null;
 30 
 31     if (viewpointParameter == j("#externalVarRelationshipViewpointParameterName").text()) {
 32         handlerFunction = odr._handleRelationshipViewGet;
 33     } else if (viewpointParameter == j("#externalVarChronologicalViewpointParameterName").text()) {
 34         handlerFunction = odr._handleChronologicalViewGet;
 35     } else if (viewpointParameter == j("#externalVarStakeholderInvolvementViewpointParameterName").text()) {
 36         handlerFunction = odr._handleStakeholderViewGet;
 37     } else {
 38         throw ("Viewpoint " + viewpointParameter + "not supported!");
 39     }
 40 
 41     j.ajax({
 42         url : requestUrl,
 43         data : parameter,
 44         dataType : "json",
 45         error : function(data, textStatus, errorThrown) {
 46             odr.setErrorText();
 47             odr.errorPopupIcon();
 48         },
 49         success : handlerFunction
 50     });
 51 
 52     
 53 });

 56 odr._handleRelationshipViewGet = function(data) {
 57 
 58     odr._requestedData = data;
 59 
 60     var allNodes = data.Nodes;
 61 
 62     for(var i = 0; i < allNodes.length; i++) {
 63         var currentNode = allNodes[i];
 64 
 65         var x = currentNode.X;
 66         var y = currentNode.Y;
 67         var name = currentNode.Version.Decision.Name;
 68         var state = currentNode.Version.State.StatusName;
 69         var visible = currentNode.Visible;
 70 
 71         if (x == 0) {
 72             x = 50;
 73         }
 74 
 75         if (y == 0) {
 76             y = 50;
 77         }
 78 
 79         var rectangle = odr.addNode(name, state, state, x, y, visible, true, odr.css.decisionClass);
 80 
 81         rectangle.value(currentNode);
 82 
 83         odr._allRectangles[currentNode.Version.Id] = rectangle;
 84     }
 85 
 86     var allRelationships = data.Associations;
 87 
 88     for(var i = 0; i < allRelationships.length; i++) {
 89         var currentRelationship = allRelationships[i];
 90 
 91         var source = odr._allRectangles[currentRelationship.Relationship.Source.Id];
 92         var target = odr._allRectangles[currentRelationship.Relationship.Target.Id];
 93         var type = currentRelationship.Relationship.Type.Name;
 94 
 95         var association = new odr.Association();
 96         if (currentRelationship.LabelX != 0 && currentRelationship.LabelY != 0) {
 97 
 98             association._labelPosition = {
 99                 x : currentRelationship.LabelX,
100                 y : currentRelationship.LabelY
101             }
102 
103         }
104         association.source(source);
105         association.target(target);
106         association.label(type);
107 
108         for(var k = 0; k < currentRelationship.Handles.length; k++) {
109             var handle = new odr.Handle();
110             handle.x(currentRelationship.Handles[k].X);
111             handle.y(currentRelationship.Handles[k].Y);
112             association.addHandle(handle);
113         }
114 
115         association.paint();
116 
117 
118 
119         association.value(currentRelationship);
120 
121         odr._allAssociations[currentRelationship.Id] = association;
122     }
123 
124     odr.hideWaitAnimation();
125 }

128 odr._handleChronologicalViewGet = function(data) {
129     odr._requestedData = data;
130 
131     var allNodes = data.Nodes;
132 
133     var stereotypeForIterations = j("#iterationStereotype").text();
134 
135     for(var i = 0; i < allNodes.length; i++) {
136         var currentNode = allNodes[i];
137 
138         var x = currentNode.X;
139         var y = currentNode.Y;
140 
141         var name = null;
142         var stereotype = null;
143         var stereotypeToShow = null;
144         var id = null;
145         var round;
146         var extraClasses = null;
147 
148         if (currentNode.Iteration != null) {
149             id = currentNode.Iteration.Id;
150             name = currentNode.Iteration.Name;
151             stereotypeToShow = currentNode.Iteration.EndDate;
152             stereotype = stereotypeForIterations;
153             round = false;
154             extraClasses = odr.css.iterationClass;
155         } else {
156             id = currentNode.Version.Id;
157             name = currentNode.Version.Decision.Name;
158             stereotypeToShow = currentNode.Version.State.StatusName;
159             stereotype = currentNode.Version.State.StatusName;
160             round = true;
161 
162             if (currentNode.Disconnected) {
163                 extraClasses = odr.css.disconnectedClass;
164             } else {
165                 extraClasses = odr.css.decisionClass;
166             }
167         }
168 
169         var visible = currentNode.Visible;
170 
171         if (x == 0) {
172             x = 50;
173         }
174 
175         if (y == 0) {
176             y = 50;
177         }
178 
179         var rectangle = odr.addNode(name, stereotype, stereotypeToShow, x, y, visible, round, extraClasses);
180 
181         rectangle.value(currentNode);
182 
183         odr._allRectangles[id] = rectangle;
184     }
185 
186     var allRelationships = data.Associations;
187 
188     for(var i = 0; i < allRelationships.length; i++) {
189         var currentRelationship = allRelationships[i];
190 
191         var sourceId;
192         var targetId;
193 
194         if (currentRelationship.SourceIteration != null) {
195             sourceId = currentRelationship.SourceIteration.Id;
196         } else {
197             sourceId = currentRelationship.SourceVersion.Id;
198         }
199 
200         if (currentRelationship.TargetIteration != null) {
201             targetId = currentRelationship.TargetIteration.Id;
202         } else {
203             targetId = currentRelationship.TargetVersion.Id;
204         }
205 
206         var source = odr._allRectangles[sourceId];
207         var target = odr._allRectangles[targetId];
208 
209         var association = new odr.Association();
210         if (currentRelationship.LabelX != 0 && currentRelationship.LabelY != 0) {
211 
212             association._labelPosition = {
213                 x : currentRelationship.LabelX,
214                 y : currentRelationship.LabelY
215             }
216 
217         }
218         
219         association.source(source);
220         association.target(target);
221 
222         for(var k = 0; k < currentRelationship.Handles.length; k++) {
223             var handle = new odr.Handle();
224             handle.x(currentRelationship.Handles[k].X);
225             handle.y(currentRelationship.Handles[k].Y);
226             association.addHandle(handle);
227         }
228 
229         association.paint();
230 
231         association.value(currentRelationship);
232 
233         odr._allAssociations[currentRelationship.Id] = association;
234     }
235 
236     odr.hideWaitAnimation();
237 }
