112 /**
113  * @description
114  * Use this utility functionality to add nodes. It has several benefits as it will make sure that the node will be
115  * added to the nodes menu and it will also add the node to the status menu.
116  *
117  * @param {String} name The name which should be shown for this node
118  * @param {String} status The status that will be visible in the status menu
119  * @param {String} statusToShow The status that will be shown above the name of the node in the node and in the decisions menu
120  * @param {Boolean} [visible] Whether this node is visible. Default = false
121  * @return {odr.Node} The new node.
122  */
123 odr.addNode = function(name, status, statusToShow, visible) {
124     if (visible == undefined) {
125         visible = false;
126     }
127 
128     var node = new odr.Node();
129     node.label(name);
130     node.status(statusToShow);
131 
132     var container = $(document.createElement("div")).
133     addClass("decisionSelector").
134     appendTo("div.decisions div.overflowDecisions").
135     append($(document.createElement("span")).text(name)).
136     append($(document.createElement("span")).text(status).addClass("stereotype"));
137 
138     var showText = odr.translation.text["menu.nodes.show"];
139     var hideText = odr.translation.text["menu.nodes.hide"];
140 
141     var currentText, currentClass;
142 
143     if (node.visible()) {
144         currentText = hideText;
145         currentClass = "hide";
146     } else {
147         currentText = showText;
148         currentClass = "show";
149     }
150 
151     container.html(container.html() + '<input type="button" class="' + currentClass + '" value="' + currentText + '"/>');
152 
153     var button = container.children("input");
154 
155     button.click(function() {
156         if (button.hasClass("hide")) {
157             button.removeClass("hide");
158             button.addClass("show");
159             button.val(showText);
160 
161             node.visible(false);
162         } else {
163             button.removeClass("show");
164             button.addClass("hide");
165             button.val(hideText);
166 
167             node.visible(true);
168         }
169     });
170 
171     node.bind(odr.Drawable.listener.visibilityChanged, function() {
172         if (node.visible()) {
173             button.removeClass("show");
174             button.addClass("hide");
175             button.val(hideText);
176         } else {
177             button.removeClass("hide");
178             button.addClass("show");
179             button.val(showText);
180         }
181     }, "nodeMenuListener");
182 
183 
184 
185 
186     if (odr.vars.statusGroups[statusToShow] == undefined) {
187         odr.vars.statusGroups[statusToShow] = [];
188         odr._newStatusGroup(statusToShow, odr.vars.statusGroups[statusToShow]);
189     }
190 
191     var statusGroup = odr.vars.statusGroups[statusToShow];
192 
193     statusGroup.push(node);
194 
195     node.bind(odr.Drawable.listener.visibilityChanged, function() {
196         var allVisible = true;
197         var allInvisible = true;
198 
199         for(var i = 1; i < statusGroup.length; i++) {
200             if (statusGroup[i].visible()) {
201                 allInvisible = false;
202             } else {
203                 allVisible = false;
204             }
205         }
206 
207         if (allVisible) {
208             statusGroup[0].attr("checked", true);
209         } else if (allInvisible) {
210             statusGroup[0].removeAttr("checked");
211         } else {
212             statusGroup[0].removeAttr("checked");
213         }
214         
215     }, "statusMenuListener");
216     
217 
218     node.visible(visible);
219 
220     return node;
221 }

227 /**
228  * @private
229  * @description
230  * Create a new status group entry in the menu
231  *
232  * @param {String} name The name of the group
233  * @param {odr.Drawable[]} items Some items which visibility is to be controled
234  */
235 odr._newStatusGroup = function(name, items) {
236     var container = $(document.createElement("div")).
237     appendTo("div.status div.overflowStatus").
238     append($(document.createElement("span")).text(name));
239 
240     container.html('<input type="checkbox" checked="true" id="stereotypeCheckbox' + name + '"/>' + container.html());
241 
242     var checkbox = container.children("input");
243 
244     items[0] = checkbox;
245 
246     checkbox.unbind().change(function() {
247         var checked = checkbox.is(":checked");
248 
249         for(var i = 1; i < items.length; i++) {
250             items[i].visible(checked);
251         }
252     });
253 }

305 /**
306  * @private
307  * @description
308  * Use this method to save all the new positions, drag handles etc.
309  */
310 odr._saveAll = function() {
311     var popupId = odr.popup.showSave();
312 
313     for(var e in odr.vars.allDecisionNodes) {
314         odr._saveNodeData(odr.vars.allDecisionNodes[e]);
315     }
316     
317     for(var e in odr.vars.allIterationNodes) {
318         odr._saveNodeData(odr.vars.allIterationNodes[e]);
319     }
320 
321     for(var i in odr.vars.allAssociations) {
322         var currentAssociation = odr.vars.allAssociations[i];
323         var json = currentAssociation.json;
324 
325         var labelPosition = currentAssociation.labelPosition();
326 
327 
328         json.LabelX = labelPosition.x;
329         json.LabelY = labelPosition.y;
330 
331         var handles = currentAssociation.handles();
332 
333         var handlesForValue = [];
334 
335         for (var k = 0; k < handles.length; k++) {
336             var currentHandle = handles[k];
337 
338             handlesForValue.push({
339                 X : currentHandle.x(),
340                 Y : currentHandle.y()
341             });
342         }
343 
344         json.Handles = handlesForValue;
345     }
346 
347     var parameter = {};
348     parameter[odr.settings.request.parameter.projectId] = odr.vars.requestParameter[odr.settings.request.parameter.projectId];
349     parameter[odr.settings.request.parameter.data] = odr._createJSONString(odr.vars.json);
350     parameter[odr.vars.requestedViewpoint] = true;
351 
352     $.ajax({
353         url : odr.settings.request.dataReceiver,
354         data : parameter,
355         dataType : "json",
356         type : "POST",
357         success : function(data, textStatus, XMLHttpRequest) {
358             odr.popup.close(popupId);
359         },
360         error : function(data, textStatus, errorThrown) {
361             odr.popup.showError("Could not save visualization: " + textStatus);
362         }
363     });
364 };

375 odr._saveNodeData = function(node) {
376     var json = node.json;

378     json.X = node.x();
379     json.Y = node.y();
380     json.Width = node.width();
381     json.Height = node.height();
382     json.Visible = node.visible();

390 /**
391  * @private
392  *
393  * @description
394  * The following function could be replaced by a JavaScript JSON serializer like
395  * JSON2 or the jquery json plugin. But for some reason, every quote is escaped.
396  *
397  * http://groups.google.com/group/google-gson/browse_thread/thread/98249309c8455a91
398  *
399  * One good thing about this approach is that only the absolutely necessary information is sent to the server.
400  *
401  * @param {Object} data The original JSON data that should be transformed
402  */
403 odr._createJSONString = function(data) {
404     var json = [];
405 
406     json.push('{"Id":',
407         data.Id,
408         ',"Nodes":[');
409 
410 
411     for (var i = 0; i < data.Nodes.length; i++) {
412         var currentNode = data.Nodes[i];
413 
414         json.push('{"Id":',
415             currentNode.Id,
416             ',"Visible":',
417             currentNode.Visible,
418             ',"X":',
419             currentNode.X,
420             ',"Y":',
421             currentNode.Y,
422             ',"Width":',
423             currentNode.Width,
424             ',"Height":',
425             currentNode.Height,
426             '}'
427             );
428 
429         if (i+1 != data.Nodes.length) {
430             json.push(',');
431         }
432     }
433 
434     json.push('],"Associations":[');
435 
436     for(var i = 0; i < data.Associations.length; i++) {
437         var currentAssociation = data.Associations[i];
438 
439         json.push('{"Id":',
440             currentAssociation.Id,
441             ',"LabelX":',
442             currentAssociation.LabelX,
443             ',"LabelY":',
444             currentAssociation.LabelY,
445             ',"Handles":[');
446 
447         for(var k = 0; k < currentAssociation.Handles.length; k++) {
448             var currentHandle = currentAssociation.Handles[k];
449 
450             json.push('{"X":',
451                 currentHandle.X,
452                 ',"Y":',
453                 currentHandle.Y,
454                 '}');
455 
456             if (k+1 != currentAssociation.Handles.length) {
457                 json.push(',');
458             }
459         }
460 
461         json.push("]}");
462 
463         if (i+1 != data.Associations.length) {
464             json.push(',');
465         }
466     }
467 
468     json.push(']}');
469 
470     return json.join("");
471 }

534 /*
535  * ###############################################################################################################
536  *                                              Exporting
537  */
538 odr.ready(function() {
539     $("div.export ul li").click(function() {
540         var format = $(this).attr("class");
541 
542         var nodes = [];
543 
544         var minX = Number.MAX_VALUE, maxX = Number.MIN_VALUE, minY = Number.MAX_VALUE, maxY = Number.MIN_VALUE;
545 
546         for(var key in odr.vars.shapesThatDetermineCanvasSize) {
547             var shape = odr.vars.shapesThatDetermineCanvasSize[key];
548 
549             if (!shape.visible() || (shape instanceof odr.Label && shape.label() == "")) {
550                 continue;
551             }
552 
553             var topLeft = shape.topLeft();
554             var bottomRight = shape.bottomRight();
555 
556             minX = Math.min(topLeft.x, minX);
557             maxX = Math.max(bottomRight.x, maxX);
558 
559             minY = Math.min(topLeft.y, minY);
560             maxY = Math.max(bottomRight.y, maxY);
561         }
562 
563         minX -= odr.settings["export"].padding.left;
564         maxX += odr.settings["export"].padding.right;
565         minY -= odr.settings["export"].padding.top;
566         maxY += odr.settings["export"].padding.bottom;
567 
568         var viewBox = ['viewBox="',
569         minX,
570         " ",
571         minY,
572         " ",
573         maxX - minX,
574         " ",
575         maxY - minY,
576         '"'].join("");
577 
578         for(var e in odr.vars.allDecisionNodes) {
579             nodes.push(odr.vars.allDecisionNodes[e].svgRepresentation());
580         }
581 
582         for(var e in odr.vars.allIterationNodes) {
583             nodes.push(odr.vars.allIterationNodes[e].svgRepresentation());
584         }
585 
586         for(var e in odr.vars.allAssociations) {
587             nodes.push(odr.vars.allAssociations[e].svgRepresentation());
588         }
589 
590         var lines = $("#" + odr.settings.svg.container).html();
591 
592         var completeSvg = lines.substring(0, lines.search("</svg>")) + nodes.join("") + lines.substring(lines.search("</svg>"));
593 
594         // the viewBox attribute needs to changed in order to cut off the sides of the svg
595         completeSvg = completeSvg.replace(/viewBox=["']{1}[0-9]+ [0-9]+ [0-9]+ [0-9]+["']{1}/, viewBox);
596         completeSvg = completeSvg.replace(/width=["']{1}[0-9]+["']{1}/, 'width="' + (maxX - minX) + '"');
597         completeSvg = completeSvg.replace(/height=["']{1}[0-9]+["']{1}/, 'height="' + (maxY - minY) + '"');
598 
599         var form = $("div.export form");
600 
601         var dataInput = form.children("input#data");
602         dataInput.val(completeSvg);
603 
604         var formatInput = form.children("input#format");
605         formatInput.val(format);
606 
607         form.submit();
608     });
609 });
